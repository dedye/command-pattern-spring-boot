package com.doku.wallet.dto.response;

import lombok.Builder;
import lombok.Data;
import org.springframework.http.HttpStatus;
import org.springframework.lang.Nullable;
import org.springframework.util.Assert;
import java.util.*;

@Data
@Builder
public class BaseResponse<T> {

    private Integer code;

    private String status;

    private T data;

    public static <T> BaseResponse<T> ok(T data) {
        Assert.notNull(data, "Data must not null");
        return BaseResponse.status(HttpStatus.OK, data);
    }

    public static <T> BaseResponse<T> okOrNotFound(@Nullable T data) {
        if (Objects.isNull(data)) {
            return BaseResponse.status(HttpStatus.NOT_FOUND, null);
        } else {
            return BaseResponse.ok(data);
        }
    }

    public static <T> BaseResponse<T> status(HttpStatus httpStatus, @Nullable T data) {
        return BaseResponse.<T>builder()
                .code(httpStatus.value())
                .status(httpStatus.getReasonPhrase())
                .data(data)
                .build();
    }

}
